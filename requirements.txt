      <label>Your name</label>
      <input id="name" type="text" placeholder="e.g. akshit" autocomplete="off">
      <label>PIN (4 digits)</label>
      <input id="pin" type="password" placeholder="1234" maxlength="6">
      <button onclick="doLogin()">Login</button>
      <div class="toggle">New here? <a onclick="toggleForm()">Create account</a></div>
    </div>
    
    <div id="register-form" style="display:none">
      <label>Choose a name</label>

      <label>Choose a PIN (4+ digits)</label>
      <input id="reg-pin" type="password" placeholder="1234" maxlength="6">
      <button onclick="doRegister()">Create account & get ₹1,00,000</button>
      <div class="toggle">Already registered? <a onclick="toggleForm()">Login</a></div>
    </div>

  <script>
    function toggleForm() {
      const l = document.getElementById('login-form');
      const r = document.getElementById('register-form');
      l.style.display = l.style.display === 'none' ? 'block' : 'none';
      r.style.display = r.style.display === 'none' ? 'block' : 'none';
    }
    function showError(msg) {
      const e = document.getElementById('error');
      e.textContent = msg; e.style.display = 'block';
    }
    async function doLogin() {
      const name = document.getElementById('name').value;
      const pin = document.getElementById('pin').value;
      const res = await fetch('/login', {
        method: 'POST', headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({name, pin})
      });
      const data = await res.json();
      if (data.success) window.location = '/dashboard';
      else showError(data.error);
    }
    async function doRegister() {
      const name = document.getElementById('reg-name').value;
      const pin = document.getElementById('reg-pin').value;
      const res = await fetch('/register', {
        method: 'POST', headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({name, pin})
      });
      const data = await res.json();
      if (data.success) window.location = '/dashboard';
      else showError(data.error);
    }
  </script>
